        switch (bit_depth) {
            case 1:
                channels_per_byte = 8;
                //pixels_per_byte = 8;
                break;
            case 2:
                channels_per_byte = 4;
                //pixels_per_byte = 4;
                break;
            case 4:
                channels_per_byte = 2;
                //pixels_per_byte = 2;
                break;
            case 8:
                channels_per_byte = 1;
                //pixels_per_byte = 1;
                break;
            default:
                channels_per_byte = 1;
                //pixels_per_byte = 1;
                break;
        }


        switch (color_type) {
            case PNG_COLOR_TYPE_GRAY:
                if(bit_depth == 8){
                    bytes_per_pixel = channels_per_byte * 1;
                }

                else
                    bytes_per_pixel = channels_per_byte * 2;
                break;

            case PNG_COLOR_TYPE_RGB:
                bytes_per_pixel = channels_per_byte * 3;
                break;


            case PNG_COLOR_TYPE_PALETTE:
                bytes_per_pixel = 1;
                break;

            case PNG_COLOR_TYPE_GRAY_ALPHA:
                bytes_per_pixel = channels_per_byte * 4;
                break;

            case PNG_COLOR_TYPE_RGB_ALPHA:
                bytes_per_pixel = channels_per_byte * 4;
                break;

            default:
                break;
        }
